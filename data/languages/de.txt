//ELLIPSIS
rules ellipsis
	rule *, .inhalt[punkt], "<.inhalt>.."
	rule *, *, "<.inhalt>..."


//NORMALE AUSSAGEN
//Normales Verhalten: Satz mit Grossbuchstaben am Anfang und Schlusspunkt.
rules aussage
	tag punkt
	//normal, hier
	rule *, .umstände.ort, "<+=:ort>{satz_verb_zuerst}."

	//normal
	rule *, *, "<+=:satz>."

	//verb am schluss
	rule verb_am_schluss, *, "{satz_verb_am_schluss}"

	//voraussetzung
	rule vorauss, .umstände.voraussetzung, " <.umstände.voraussetzung>"
	rule vorauss, *, ""

	//zeitangabe
	rule zeitangabe, .umstände.zeit[anzeigen], " <.umstände.zeit>"
	rule zeitangabe, *, ""

	//zeitangabe am anfang
	//rule zeitanf, .umstände.zeit[insatz], "<.umstände.zeit> "
	rule zeitanf, *, ""

	//Perfekt
	rule satz, .umstände.zeit[vergangenheit_mit_auswirkung_auf_gegenwart], "{zeitanf}<.aktion:perf>{vorauss}{zeitangabe}"
	rule satz_verb_zuerst, .umstände.zeit[vergangenheit_mit_auswirkung_auf_gegenwart], "{zeitanf}<.aktion:perf_verb_zuerst>{vorauss}{zeitangabe}"
	rule satz_verb_am_schluss, .umstände.zeit[vergangenheit_mit_auswirkung_auf_gegenwart], "{zeitanf}<.aktion:perf_verb_am_schluss_vor_ort>{ort}<.aktion:perf_verb_am_schluss_nach_ort>{vorauss}{zeitangabe}"

	//normal
	rule satz, *, "{zeitanf}<.aktion>{vorauss}{zeitangabe}"
	rule satz_verb_zuerst, *, "{zeitanf}<.aktion:verb_zuerst>{vorauss}{zeitangabe}"
	rule satz_verb_am_schluss, *, "{zeitanf}<.aktion:verb_am_schluss_vor_ort>{ort}<.aktion:verb_am_schluss_nach_ort>{vorauss}{zeitangabe}"

	//ort
	rule ort, .umstände.ort, "<.umstände.ort> "
	rule ort, *, ""

//VERB
rules verb

	//NORMAL
	//normal, passiv
	rule *, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}{werden_ziel}{_z_insatz}{negation} {adv_}{v_partizip}{für}"
	//normal, aktiv
	rule *, .akteur, "{akteur} {v_präsens}{_z_insatz}{_ziel}{negation}{_adv}{für}{_nach_objekt}"
	//infinitiv
	rule *, *, "{z_insatz_}{negation_}{adv_}{infinitiv}"

	//normal, verb zuerst, passiv
	rule verb_zuerst, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{werden_ziel}{_z_insatz}{_ziel}{negation} {adv_}{v_partizip}{für}"
	//normal, verb zuerst, aktiv
	rule verb_zuerst, .akteur, "{v_präsens}{_z_insatz} {akteur}{_ziel}{negation}{_adv}{für}{_nach_objekt}"
	//verb zuerst, infinitiv
	rule verb_zuerst, *, "{z_insatz_}{negation_}{adv_}{infinitiv}"

	//normal, kein akteur, passiv
	rule kein_akteur, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}{werden_ziel}{_z_insatz}{negation} {adv_}{v_partizip}{für}"
	//normal, kein akteur, aktiv
	rule kein_akteur, .akteur, "{v_präsens}{_z_insatz}{negation}{_ziel}{_adv}{für}{_nach_objekt}"
	//kein akteur, infinitiv
	rule kein_akteur, *, "{z_insatz_}{negation_}{adv_}{infinitiv}"

	//verb am schluss
	rule verb_am_schluss_vor_ort, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}"
	rule verb_am_schluss_nach_ort, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{z_insatz_}{negation_}{für_}{adv_}{v_partizip} {werden_ziel}"
	rule verb_am_schluss_vor_ort, .akteur, "{akteur} "
	rule verb_am_schluss_nach_ort, .akteur, "{z_insatz_}{negation_}{ziel_}{für_}{nach_objekt_}{adv_}{v_präsens}"
	rule verb_am_schluss_vor_ort, *, ""
	rule verb_am_schluss_nach_ort, *, "{z_insatz_}{negation_}{für_}{adv_}{infinitiv}"


	//Perfekt
	//Perfekt, passiv
	rule perf, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}{sein_ziel}{_z_insatz}{negation}{_adv}{für} {v_partizip} worden"
	//Perfekt, aktiv
	rule perf, .akteur, "{akteur} {haben_akteur}{_ziel}{_z_insatz}{negation}{_adv}{für} {v_partizip}"
	//quasi infinitiv
	rule perf, *, "{z_insatz_}{negation_}{adv_}{v_partizip} haben"

	//Perfekt, verb zuerst, passiv
	rule perf_verb_zuerst, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{sein_ziel}{_z_insatz}{_ziel}{negation}{_adv}{für} {v_partizip} worden"
	//Perfekt, verb zuerst, aktiv
	rule perf_verb_zuerst, .akteur, "{haben_akteur}{_z_insatz} {akteur}{_ziel}{negation}{_adv}{für} {v_partizip}"
	//quasi infinitiv, verb zuerst
	rule perf_verb_zuerst, *, "{z_insatz_}{negation_}{adv_}{v_partizip} haben"

	//Perfekt, kein akteur, passiv
	rule perf_kein_akteur, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}{sein_ziel}{_z_insatz}{negation}{_adv}{für} {v_partizip} worden"
	//Perfekt, kein akteur, aktiv
	rule perf_kein_akteur, .akteur, "{haben_akteur}{_z_insatz}{_ziel}{negation}{_adv}{für} {v_partizip}"
	//quasi infinitiv, kein akteur
	rule perf_kein_akteur, *, "{z_insatz_}{negation_}{adv_}{v_partizip} haben"

	//verb am schluss
	rule perf_verb_am_schluss_vor_ort, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}"
	rule perf_verb_am_schluss_nach_ort, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{z_insatz_}{negation_}{adv_}{für_}{v_partizip} worden {sein_ziel}"
	rule perf_verb_am_schluss_vor_ort, .akteur, "{akteur} {ziel_}"
	rule perf_verb_am_schluss_nach_ort, .akteur, "{z_insatz_}{negation_}{adv_}{für_}{v_partizip} {haben_akteur}"
	rule perf_verb_am_schluss_vor_ort, *, ""
	rule perf_verb_am_schluss_nach_ort, *, "{z_insatz_}{negation_}{für_}{adv_}{v_partizip} haben"




	//FÄHIG
	//fähig, passiv
	rule fähig, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}{können_ziel}{_z_insatz}{negation}{_adv}{für} {v_partizip} werden"

	rule fähig, .akteur, "{akteur} {können_akteur}{_z_insatz}{negation}{_ziel}{_adv}{für} {infinitiv}"
	//fähig, infinitiv
	rule fähig, *, "{adv_}{infinitiv}{_z_insatz}{negation} können"

	rule fähig_verb_zuerst, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{können_ziel}{_z_insatz}{negation}{_ziel}{_adv}{für} {v_partizip} werden"
	//normal, verb zuerst, aktiv
	rule fähig_verb_zuerst, .akteur, "{können_akteur}{_z_insatz} {akteur}{negation}{_ziel}{_adv}{für} {infinitiv}"
	//verb zuerst, infinitiv
	rule fähig_verb_zuerst, *, "{adv_}{infinitiv}{_z_insatz}{negation} können"

	//fähig, kein akteur, passiv
	rule fähig_kein_akteur, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}{können_ziel}{_z_insatz}{negation}{für} {adv_}{v_partizip} werden"
	//fähig, aktiv
	rule fähig_kein_akteur, .akteur, "{können_akteur}{_z_insatz}{negation}{_ziel}{für} {adv_}{infinitiv}"
	//fähig, infinitiv
	rule fähig_kein_akteur, *, "{adv_}{infinitiv}{_z_insatz}{negation} können"

	//Perfekt, FÄHIG
	//Perfekt, passiv
	rule perf_fähig, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}{konnte_ziel}{_z_insatz}{negation}{für} {adv_}{v_partizip} werden"

	//quasi infinitiv
	rule perf_fähig, *, "{z_insatz_}{negation_}{adv_}{v_partizip} haben können"

	//Perfekt, verb zuerst, passiv
	rule perf_fähig_verb_zuerst, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{konnte_ziel}{_z_insatz}{_ziel}{negation}{für} {adv_}{v_partizip} werden"
	//Perfekt, verb zuerst, aktiv
	rule perf_fähig_verb_zuerst, .akteur, "{haben_akteur}{_z_insatz} {akteur}{_ziel}{negation}{für} {adv_}{infinitiv} können"
	//quasi infinitiv, verb zuerst
	rule perf_fähig_verb_zuerst, *, "{z_insatz_}{negation_}{adv_}{v_partizip} haben können"

	//Perfekt, kein akteur, passiv
	rule perf_fähig_kein_akteur, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}{konnte_ziel}{_z_insatz}{negation}{für} {adv_}{v_partizip} worden"
	//Perfekt, kein akteur, aktiv
	rule perf_fähig_kein_akteur, .akteur, "{haben_akteur}{_z_insatz}{_ziel}{negation}{für} {adv_}{infinitiv} können"
	//quasi infinitiv, kein akteur
	rule perf_fähig_kein_akteur, *, "{z_insatz_}{negation_}{adv_}{v_partizip} haben können"

	rule perf_fähig_verb_am_schluss_vor_ort, (!(.akteur) & (.ziel | ^.aktion1.ziel)), "{ziel_}"

//ADJEKTIVE
rules adjektiv
	//verändert von kasus, numerus, geschlecht, artikel
	//wobei der plural quasi ein 4. geschlecht ist

	rule *, *, "{stamm}"

	//nominativ bestimmt ("spezifisch")
	rule nom, (^.anzahl[plural] & (^[spezifisch] | ^.welcher[spezifisch] | ^.anzahl[spezifisch])), "{adv}{stamm}en"
	rule nom, (^[spezifisch] | ^.welcher[spezifisch] | ^.anzahl[spezifisch]), "{adv}{stamm}e"

	//nominativ unbestimmt
	rule nom, ^.anzahl[plural], "{adv}{stamm}e"
	rule nom, ^[säch], "{adv}{stamm}es"
	rule nom, ^[weib], "{adv}{stamm}e"
	rule nom, *, "{adv}{stamm}er"

	//akkusativ bestimmt
	rule akk, (^.anzahl[plural] & (^[spezifisch] | ^.welcher[spezifisch] | ^.anzahl[spezifisch])), "{adv}{stamm}en"
	rule akk, (^[säch] & (^[spezifisch] | ^.welcher[spezifisch] | ^.anzahl[spezifisch])), "{adv}{stamm}e"
	rule akk, (^[weib] & (^[spezifisch] | ^.welcher[spezifisch] | ^.anzahl[spezifisch])), "{adv}{stamm}e"
	rule akk, (^[spezifisch] | ^.welcher[spezifisch] | ^.anzahl[spezifisch]), "{adv}{stamm}en"

	//akkusativ unbestimmt
	rule akk, ^.anzahl[plural], "{adv}{stamm}e"
	rule akk, ^[säch], "{adv}{stamm}es"
	rule akk, ^[weib], "{adv}{stamm}e"
	rule akk, *, "{adv}{stamm}en"

	//dativ
	rule dat, *, "{adv}{stamm}en"

	//infinitiv sozusagen
	rule *, *, "{adv}{stamm}"

	//adverb
	rule adv, .modifikator, "<.modifikator> "
	rule adv, *, ""

	rule _adv, .modifikator, " <.modifikator>"
	rule _adv, *, ""
